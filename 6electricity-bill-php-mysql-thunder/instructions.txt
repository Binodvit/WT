CREATE TABLE Consumer (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100),

  address VARCHAR(255)
);

CREATE TABLE Billing (
  id INT AUTO_INCREMENT PRIMARY KEY,
  consumer_id INT,
  units INT,
  amount DECIMAL(10,2),
  billing_date DATE,
  FOREIGN KEY (consumer_id) REFERENCES Consumer(id)
);
